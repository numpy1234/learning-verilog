    m=1'b0;

Behavioural description of 4 to 1 line mux

module mux4x1(i0, i1, i2, i3, select, y);
  input i0, i1, i2, i3;
  input [1:0] select;
  output y;
  reg y;
  always@(i0 or i1 or i2 oe i3 or select) ///executes when i0, i1, i2, i3 or select changes 
  begin
    case(select)
      2'b00: y= i0;
      2'b01: y= i1;
      2'b10: y= i2;
      2'b11: y=i3;
    endcase
  end
endmodule

= : blocking, executed in the order they are specified in a sequential block, will not block execution of statatements that follow in a parallel block
<=: non blocking allow scheduling of assignments without blocking execution of statements that follow in a sequential block

module block();
reg a,b,c,d,e,f;
initial 
begin
  a= #10 1'b1;     //assigns 1 to a at time 10
  b= #20 1'b0;     //assigns 0 to b at time 30 (10+20)
  a= #40 1'b1;     //assigns 1 to c at time 70 (30+40)
end
endmodule

these proess run sequentially and wait for previous to finish





module non_block();
initial
begin
  d<= #10 1'b1; //assigns 1 to d at time 10
  e<= #20 1'b0; //assigns 0 to e at time 20 
  f<= #40 1'b1; //assigns 1 to f at time 40
end
endmodule

these processes run simultaenously and doesnt wait for previous and works on the value which was present earlier
